cd /opt
git clone https://github.com/termux-pacman/glibc-packages.git
# 克隆仓库并切换到 v17.0 标签
git clone https://github.com/pulseaudio/pulseaudio.git
cd pulseaudio
git checkout v17.0

PREFIX=/data/data/com.winlator/files/imagefs/usr
TMPDIR=/data/data/com.winlator/files/imagefs/tmp
PATCH_DIR=/opt/glibc-packages/gpkg/libpulse   # 补丁文件所在目录

# 应用路径补丁：将 Termux 变量替换为实际路径，并打补丁
if [ -f ${PATCH_DIR}/fix-paths.patch ]; then
    sed -i "s|@TERMUX_PREFIX@|${PREFIX}|g; s|@TERMUX_PREFIX_CLASSICAL@|${TMPDIR}|g" \
        ${PATCH_DIR}/fix-paths.patch
    patch -p1 < ${PATCH_DIR}/fix-paths.patch
fi

# 创建 Meson 交叉编译文件
cat > cross-aarch64.txt << 'EOF'
[binaries]
c = 'aarch64-linux-gnu-gcc'
cpp = 'aarch64-linux-gnu-g++'
ar = 'aarch64-linux-gnu-ar'
strip = 'aarch64-linux-gnu-strip'
pkgconfig = 'pkg-config'

[host_machine]
system = 'linux'
cpu_family = 'aarch64'
cpu = 'aarch64'
endian = 'little'
EOF

# 设置 pkg-config 搜索路径
export PKG_CONFIG_LIBDIR=${PREFIX}/lib/pkgconfig:${PREFIX}/share/pkgconfig
export PKG_CONFIG_SYSROOT_DIR=/data/data/com.winlator/files/imagefs

# Meson 配置
meson setup builddir \
    --cross-file cross-aarch64.txt \
    --prefix=/data/data/com.winlator/files/imagefs/usr \
    -Ddaemon=false \
    -Dman=false \
    -Dtests=false \
    -Ddatabase=gdbm \
    -Dx11=enabled \
    -Ddbus=enabled \
    -Dalsa=enabled \
    -Ddoxygen=false
# 编译 & 安装
meson compile -C builddir
meson install -C builddir

# 可选：安装 pulseaudio-start 脚本 (如果配方目录有该文件)
if [ -f ${PATCH_DIR}/pulseaudio-start.sh ]; then
    install -m755 ${PATCH_DIR}/pulseaudio-start.sh ${PREFIX}/bin/pulseaudio-start
    sed -i -e "s|@TERMUX_PREFIX_CLASSICAL@|${TMPDIR}|g" \
           -e "s|@TERMUX_PREFIX@|${PREFIX}|g" \
           ${PREFIX}/bin/pulseaudio-start
fi
cd ..
