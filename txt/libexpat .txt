# 1. 下载并解压
wget https://github.com/libexpat/libexpat/releases/download/R_2_6_4/expat-2.6.4.tar.bz2
tar -xf expat-2.6.4.tar.bz2
cd expat-2.6.4

# 2. 配置 (--without-docbook 跳过文档生成，减少依赖)
./configure \
    --host=aarch64-linux-gnu \
    --build=x86_64-linux-gnu \
    --prefix=/data/data/com.winlator/files/imagefs/usr \
    --disable-static \
    --without-docbook

# 3. 编译 & 安装 (防宿主 strip)
make -j$(nproc)
make install INSTALL_PROGRAM="/usr/bin/install -c"
cd ..
