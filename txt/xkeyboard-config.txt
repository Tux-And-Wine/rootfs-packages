wget https://xorg.freedesktop.org/archive/individual/data/xkeyboard-config/xkeyboard-config-2.43.tar.xz
tar -xf xkeyboard-config-2.43.tar.xz
cd xkeyboard-config-2.43

# 配置，xkb-base 指向 rootfs 内的路径，开启兼容规则和符号链接
meson setup builddir \
    --prefix=/data/data/com.winlator/files/imagefs/usr \
    -Dxkb-base=/data/data/com.winlator/files/imagefs/usr/share/X11/xkb \
    -Dcompat-rules=true \
    -Dxorg-rules-symlinks=true

meson compile -C builddir
meson install -C builddir
cd ..
