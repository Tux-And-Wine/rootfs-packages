wget https://github.com/libsndfile/libsndfile/archive/refs/tags/1.2.2.tar.gz -O libsndfile-1.2.2.tar.gz
tar -xzf libsndfile-1.2.2.tar.gz
cd libsndfile-1.2.2

# 打补丁：将临时目录路径改为安卓 rootfs 内的 /tmp
sed -i 's|@TERMUX_PREFIX_CLASSICAL@/tmp|/data/data/com.winlator/files/imagefs/tmp|' src/common.c

# 创建 CMake 交叉工具链文件
cat > toolchain-aarch64.cmake << 'EOF'
set(CMAKE_SYSTEM_NAME Linux)
set(CMAKE_SYSTEM_PROCESSOR aarch64)
set(CMAKE_SYSROOT /data/data/com.winlator/files/imagefs)
set(CMAKE_C_COMPILER aarch64-linux-gnu-gcc)
set(CMAKE_CXX_COMPILER aarch64-linux-gnu-g++)
set(CMAKE_FIND_ROOT_PATH /data/data/com.winlator/files/imagefs/usr)
set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
EOF

export PKG_CONFIG_LIBDIR=/data/data/com.winlator/files/imagefs/usr/lib/pkgconfig:/data/data/com.winlator/files/imagefs/usr/share/pkgconfig
export PKG_CONFIG_SYSROOT_DIR=/data/data/com.winlator/files/imagefs

cmake -B build \
    -DCMAKE_TOOLCHAIN_FILE=toolchain-aarch64.cmake \
    -DCMAKE_INSTALL_PREFIX=/data/data/com.winlator/files/imagefs/usr \
    -DBUILD_SHARED_LIBS=ON \
    -DENABLE_EXTERNAL_LIBS=ON \
    -DENABLE_MPEG=ON

cmake --build build --parallel $(nproc)
cmake --install build
cd ..
