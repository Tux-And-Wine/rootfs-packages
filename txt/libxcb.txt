wget https://xorg.freedesktop.org/archive/individual/lib/libxcb-1.17.0.tar.xz
tar -xf libxcb-1.17.0.tar.xz
cd libxcb-1.17.0

# 安卓路径补丁：Unix socket 路径改为安卓临时目录
sed -i 's|/tmp/.X11-unix/X|/data/data/com.winlator/files/imagefs/tmp/.X11-unix/X|g' src/xcb_util.c

# 设置 Python 搜索路径，让宿主 python 能找到 xcbgen 模块
export PYTHONPATH=/data/data/com.winlator/files/imagefs/usr/lib/python3/dist-packages:$PYTHONPATH

# 配置
./configure \
    --host=aarch64-linux-gnu \
    --build=x86_64-linux-gnu \
    --prefix=/data/data/com.winlator/files/imagefs/usr \
    --disable-static \
    --enable-xinput \
    --enable-xkb \
    CPPFLAGS="-I/data/data/com.winlator/files/imagefs/usr/include" \
    LDFLAGS="-L/data/data/com.winlator/files/imagefs/usr/lib"

# 编译 & 安装（防宿主 strip）
make -j$(nproc)
make install INSTALL_PROGRAM="/usr/bin/install -c"
