git clone https://github.com/ptitSeb/box64.git
cd box64
mkdir build && cd build

export LDFLAGS="-s -Wl,--dynamic-linker=/data/data/com.winlator/files/imagefs/usr/lib/ld-linux-aarch64.so.1 -Wl,-rpath=/data/data/com.winlator/files/imagefs/usr/lib"


cmake .. \
    -DCMAKE_BUILD_TYPE=RelWithDebInfo \
    -DARM64=ON \
    -DARM_DYNAREC=ON \
    -DBAD_SIGNAL=ON \
    -DWINLATOR_GLIBC=ON \
    -DCMAKE_C_COMPILER=aarch64-linux-gnu-gcc \
    -DCMAKE_CXX_COMPILER=aarch64-linux-gnu-g++ \
    -DCMAKE_SYSTEM_NAME=Linux \
    -DCMAKE_SYSTEM_PROCESSOR=aarch64 \
    -DCMAKE_CROSSCOMPILING=TRUE


make -j$(nproc)
make install DESTDIR=$PWD/package

# 编译好后请手动将package中的box64-x86_64-linux-gnu移动至目标rootfs中
