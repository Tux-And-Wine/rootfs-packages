wget https://ftp.gnu.org/gnu/ncurses/ncurses-6.5.tar.gz

tar -xzf ncurses-6.5.tar.gz
cd ncurses-6.5

# 编译参数基本不变，移除源码中不存在的 --with-cxx-shared 选项
./configure \
--host=aarch64-linux-gnu \
--build=x86_64-linux-gnu \
--prefix=/data/data/com.winlator/files/imagefs/usr \
--disable-root-access \
--disable-root-environ \
--disable-setuid-environ \
--enable-widec \
--enable-pc-files \
--with-shared \
--with-cxx-binding \
--with-versioned-syms \
--with-pkg-config-libdir=/data/data/com.winlator/files/imagefs/usr/lib/pkgconfig \
--with-xterm-kbs=del \
--without-ada \
--without-strip \
--mandir=/data/data/com.winlator/files/imagefs/usr/share/man

make -j$(nproc)
make install 


sed -i 's/^INSTALL_PROG\s*=.*/INSTALL_PROG = \$(INSTALL)/' progs/Makefile




DEST=/data/data/com.winlator/files/imagefs
PREF=$DEST/usr

for lib in ncurses ncurses++ form panel menu; do
    printf "INPUT(-l%sw)\n" "${lib}" > $PREF/lib/lib${lib}.so
    ln -svf ${lib}w.pc $PREF/lib/pkgconfig/${lib}.pc
done

printf 'INPUT(-lncursesw)\n' > $PREF/lib/libcursesw.so
ln -svf libncurses.so $PREF/lib/libcurses.so

VER_MAJOR=6
for lib in tic tinfo; do
    printf "INPUT(libncursesw.so.%s)\n" "$VER_MAJOR" > $PREF/lib/lib${lib}.so
    ln -svf libncursesw.so.${VER_MAJOR} $PREF/lib/lib${lib}.so.${VER_MAJOR}
    ln -svf ncursesw.pc $PREF/lib/pkgconfig/${lib}.pc
done

mkdir -p $PREF/include/ncurses
for i in $PREF/include/ncursesw/*; do
    mv ${i} $PREF/include/
    ln -s ../$(basename ${i}) $PREF/include/ncurses/
    ln -s ../$(basename ${i}) $PREF/include/ncursesw/
done
